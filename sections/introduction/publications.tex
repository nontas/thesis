\section{Publications}
In this section, we provide a list of publications that were authored during
the course of this Ph.D. thesis. We split these publications in two categories: \emph{(i)}~those that are related to the contents of this thesis~\ref{sec:related_publications} and \emph{(ii)}~other publications that are not directly relevant~\ref{sec:other_publications}.

\subsection{Related Publications}\label{sec:related_publications}
The work presented in this thesis is directly related to the following
publications:
\begin{itemize}
  \item \textbf{E. Antonakos}, and S. Zafeiriou.
  ``Automatic Construction of Deformable Models In-The-Wild'',
  \emph{Proceedings of IEEE International Conference on Computer Vision and Pattern Recognition (CVPR)},
  Columbus, OH, USA, pp. 1813-1820, June 2014.

  \item \textbf{E. Antonakos}, J. Alabort-i-Medina, G. Tzimiropoulos, and S. Zafeiriou.
  ``HOG Active Appearance Models'',
  \emph{Proceedings of IEEE International Conference on Image Processing (ICIP)},
  Paris, France, pp. 224-228, October 2014.

  \item J. Alabort-i-Medina\footnote{Joint first authorship.\label{equal_authors}}, \textbf{E. Antonakos}\textsuperscript{\ref{equal_authors}}, J. Booth\textsuperscript{\ref{equal_authors}}, P. Snape\textsuperscript{\ref{equal_authors}}, and S. Zafeiriou.
  ``Menpo: A Comprehensive Platform for Parametric Image Alignment and Visual Deformable Models'',
  \emph{Proceedings of ACM International Conference on Multimedia (ACMM)},
  Orlando, FL, USA, pp. 679-682, November 2014.

  \item \textbf{E. Antonakos}, J. Alabort-i-Medina, G. Tzimiropoulos, and S. Zafeiriou.

  \item \textbf{E. Antonakos}\textsuperscript{\ref{equal_authors}}, P. Snape\textsuperscript{\ref{equal_authors}}, G. Trigeorgis, and S. Zafeiriou.
  ``Adaptive Cascaded Regression'',
